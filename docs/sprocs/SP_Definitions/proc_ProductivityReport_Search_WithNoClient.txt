CREATE PROC [dbo].[proc_ProductivityReport_Search_WithNoClient]        
-- [proc_ProductivityReport_Search_WithNoClient] 'USERNAME ASC',100,0,'TH.TrxType=1 and TH.TrxStatus=200 ','daily','2015-01-01','2015-01-07','admin'        
@sortExpression VARCHAR(100) = 'USERNAME ASC',        
@maximumRows INT = 50,        
@startRowIndex INT = 0,        
@searchString VARCHAR(max) = 'TH.TrxType=1',        
@Type VARCHAR(10) = 'daily',        
@Startdate VARCHAR(50) = '2014-11-19',        
@EndDate VARCHAR(50) = '2014-12-18',        
@SessionUserCode VARCHAR(50) = '',  
@ChannelCodes VARCHAR(max) = '',    
@SalesTeamCodes VARCHAR(max)=''           
AS        
--SET @startRowIndex=@startRowIndex+1        
IF @StartDate = ''        
SET @StartDate = CAST(GETDATE() AS VARCHAR)        
IF @EndDate = ''        
SET @EndDate = CAST(GETDATE() AS VARCHAR)        
CREATE TABLE #Users(Id1 INT IDENTITY(1,1),UserCode VARCHAR(50), UserName VARCHAR(200), CallsMade INT,        
ProductiveCalls INT)    
  
--IF LEN(@searchString) = 0        
SET @searchString = ' 1 = 1 '        
INSERT INTO #Users(UserCode, UserName, CallsMade, ProductiveCalls)        
Select DISTINCT U.Code, U.UserName,0,0 from dbo.udf_GetAllChildUsers(@SessionUserCode,'ALL') UI        
Inner Join tblDailYJourneyPlan DJP On UI.UserCode=DJP.UserCode        
AND DATEDIFF(DD,@StartDate,DJP.JourneyDate) >= 0 AND DATEDIFF(DD,@EndDate,DJP.JourneyDate) <= 0        
Inner Join tblUSER U On UI.UserCode=U.Code AND U.IsActive = 1        
      
Update UA SET UA.CallsMade=Res.CallsMade, UA.ProductiveCalls=Res.ProductiveCalls from #Users UA        
INNER JOIN (        
SELECT U.Code, U.USERNAME, SUM(CallsMade) CallsMade,        
SUM(ProductiveCalls) AS ProductiveCalls        
FROM vw_UserVisitsWithChannel U        
INNER JOIN dbo.udf_GetAllChildUsers(@SessionUserCode,'All') CU On U.Code = CU.UserCode  
AND (@ChannelCodes = '' OR U.ChannelCode IN(SELECT Item FROM ufnSplit(@ChannelCodes, ',')))  
AND U.Code in (Select distinct vC.UserCode from vw_GetUserClients vC       
      INNER JOIN tblUserItemGRoup UIG ON UIG.UserCode=vC.UserCode AND (@SalesTeamCodes = '' OR UIG.SalesteamCode in (Select items from dbo.split(@SalesTeamCodes,','))))         
WHERE       
DATEDIFF(DD, @StartDate,VisitDate) >= 0        
AND DATEDIFF(DD,@EndDate,VisitDate) <= 0         
GROUP BY U.USERNAME, U.Code ) RES On UA.UserCode=Res.Code       
       
SELECT * FROM        
(        
SELECT ROW_NUMBER() OVER(ORDER BY UserCode DESC) as Id, * FROM #Users        
) AS U        
WHERE (@maximumRows = 0) OR (Id BETWEEN @startRowIndex AND (@startRowIndex + @maximumRows) - 1)        
ORDER BY UserName ASC        
      
SELECT U.UserCode AS UserCode, U.USERNAME, SUM(PlannedCalls) PlannedCalls ,SUM(ProductiveCalls) AS ProductivePlannedCalls        
FROM vw_UserPlannedVisitWithChannel U        
INNER JOIN dbo.udf_GetAllChildUsers(@SessionUserCode,'All') CU On U.UserCode = CU.UserCode  
AND (@ChannelCodes = '' OR U.ChannelCode IN(SELECT Item FROM ufnSplit(@ChannelCodes, ',')))  
AND U.UserCode in (Select distinct vC.UserCode from vw_GetUserClients vC       
      INNER JOIN tblUserItemGRoup UIG ON UIG.UserCode=vC.UserCode AND (@SalesTeamCodes = '' OR UIG.SalesteamCode in (Select items from dbo.split(@SalesTeamCodes,','))))        
WHERE DATEDIFF(DD, @StartDate,VisitDate) >= 0        
AND DATEDIFF(DD,@EndDate,VisitDate) <= 0       
GROUP BY U.USERNAME, U.UserCode       
      
select COUNT(1) AS ProductivityCount from #Users      
      
DROP TABLE #Users
