Select                                               
CASE WHEN TH.TrxType=4 THEN -1 ELSE 1 END *                                  
ROUND(((( ISNULL(TD.QuantityLevel2,0) * CASE WHEN TH.TrxType=12 THEN  ISNULL(TD.PriceUsedLevel3,0) ELSE ISNULL(TD.PriceUsedLevel2,0) END ) +                                  
 (ISNULL(TD.QuantityLevel3,0) * CASE WHEN TH.TrxType=12 THEN  ISNULL(TD.PriceUsedLevel2,0) ELSE ISNULL(TD.PriceUsedLevel3,0) END ))

,0) AS  SalesAmount ,                                                    
Case When (TD.ItemType='R' OR TH.TrxType=4 ) Then (TD.PriceUsedLevel1 * TD.QuantityLevel1) ELSE 0 END AS ReturnAmount       
from #Trx TH With(NoLock)                        
Inner Join tblTrxDetail TD With(NoLock) ON                       
TH.TrxCode=TD.TrxCode              
                                      
Inner Join tblItem I With(NoLock) ON I.Code=TD.ItemCode AND I.SalesOrgCode=TH.OrgCode                       
Inner JOIN tblItemGroup CA WITH(NOLOCK) ON CA.Code=I.GroupLevel3 AND CA.ItemGroupLevelid=2

 ) as T-- Sales & Movement
