CREATE PROC [dbo].[proc_Report_EffeciencyReport_ByUser]  
(                                                       
@Startdate VARCHAR(50) = '2015-02-01',                                                      
@EndDate VARCHAR(50) = '2015-02-17',                    
@SessionUserCode VARCHAR(50) = 'admin' ,  
@ChannelCodes VARCHAR(max) = '',      
@SalesTeamCodes VARCHAR(max)=''  
)  
AS  
IF @StartDate = ''                                                          
SET @StartDate = CAST(GETDATE() AS VARCHAR)                                   
                                                
IF @EndDate = ''                                                          
SET @EndDate = CAST(GETDATE() AS VARCHAR)   
  
DECLARE @TemTable TABLE(Id INT IDENTITY(1,1), UserCode VARCHAR(100),UserName VARCHAR(200),ProductiveCalls INT,AvgValuePerPC FLOAT, LinesperPC FLOAT,MaxLineinPC INT, MinLineinPC INT)                                                                        
  
INSERT INTO @TemTable(UserCode,UserName,ProductiveCalls,AvgValuePerPC, LinesperPC,MaxLineinPC, MinLineinPC)   
Select DISTINCT U.Code, U.UserName,0,0,0,0,0 from dbo.udf_GetAllChildUsers(@SessionUserCode,'ALL') UI    
INNER JOIN tblDailYJourneyPlan DJP On UI.UserCode=DJP.UserCode          
AND DATEDIFF(DD,@StartDate,DJP.JourneyDate) >= 0 AND DATEDIFF(DD,@EndDate,DJP.JourneyDate) <= 0            
INNER JOIN tblUSER U On UI.UserCode=U.Code AND U.IsActive = 1  
AND U.Code in (Select distinct vC.UserCode from vw_GetUserClients vC         
   INNER JOIN tblUserItemGRoup UIG ON UIG.UserCode=vC.UserCode AND (@SalesTeamCodes = '' OR UIG.SalesteamCode in   
        (Select items from dbo.split(@SalesTeamCodes,','))))  
          
DECLARE @TempLines TABLE(TrxCode VARCHAR(50), LineCount INT)   
  
DECLARE @Count INT = 0, @TotalRows INT, @CurrId INT = 1, @UserCode VARCHAR(50)  
  
SELECT @TotalRows = COUNT(1) FROM @TemTable  
  
WHILE(@Count < @TotalRows)  
BEGIN  
   
 DELETE FROM @TempLines  
   
 SELECT @UserCode = UserCode FROM @TemTable WHERE Id = @CurrId  
   
 DECLARE @ProductiveCalls INT = 0, @AvgValuePerPC FLOAT = 0, @LinesperPC FLOAT = 0  
   
 SELECT @ProductiveCalls = Sum(UV.ProductiveCalls)   
 FROM vw_UserVisitsWithChannel_Productivity UV  
 WHERE (@ChannelCodes = '' OR UV.ChannelCode IN(SELECT Item FROM ufnSplit(@ChannelCodes, ',')))  
 AND DATEDIFF(DD, @StartDate, VisitDate) >= 0  AND  DATEDIFF(DD, @EndDate, VisitDate) <= 0  
 AND UV.Code = @UserCode   
 GROUP BY UV.Code  
   
 SElECT @AvgValuePerPC = SUM(ISNULL(TH.TotalAmount,0) - ISNULL(TH.TotalDiscountAmount,0))  
 FROM tblCustomerVisit V           
 INNER JOIN tblCustomer C ON C.Code = V.ClientCode AND V.UserCode = @UserCode                
 INNER JOIN tblCustomerDetail CD On C.Code =CD.CustomerCode   
  AND (@ChannelCodes = '' OR CD.ChannelCode IN(SELECT Item FROM ufnSplit(@ChannelCodes, ',')))                 
 INNER JOIN tblTrxHeader TH ON TH.UserCode = V.UserCode AND TH.VisitCode = V.VisitCode                  
  AND V.ClientCode = TH.ClientCode  
  AND TH.OrgCode = CD.SalesOrgCode       
  AND TH.TrxType IN (1,3,5) AND TH.TrxStatus IN (100,200,3)            
  AND DATEDIFF(dd, TH.TrxDate, V.Date) = 0  
 WHERE (DATEDIFF(dd,@StartDate,V.[Date]) >= 0 AND DATEDIFF(dd,@EndDate,V.[Date]) <= 0)                 
 GROUP BY V.UserCode  
   
 --SElECT @LinesperPC = COUNT(TD.[LineNo])  
 --FROM tblCustomerVisit V           
 --INNER JOIN tblCustomer C ON C.Code = V.ClientCode AND V.UserCode = @UserCode                
 --INNER JOIN tblCustomerDetail CD On C.Code =CD.CustomerCode   
 -- AND (@ChannelCodes = '' OR CD.ChannelCode IN(SELECT Item FROM ufnSplit(@ChannelCodes, ',')))                 
 --INNER JOIN tblTrxHeader TH ON TH.UserCode = V.UserCode AND TH.VisitCode = V.VisitCode                  
 -- AND V.ClientCode = TH.ClientCode  
 -- AND TH.OrgCode = CD.SalesOrgCode       
 -- AND TH.TrxType IN (1,3,5) AND TH.TrxStatus IN (100,200,3)-- AND DATEDIFF(dd, TH.TrxDate, V.Date) = 0  
 --INNER JOIN tblTrxDetail TD ON TH.TrxCode = TD.TrxCode  
 --WHERE (DATEDIFF(dd,@StartDate,V.[Date]) >= 0 AND DATEDIFF(dd,@EndDate,V.[Date]) <= 0)                 
 --GROUP BY TH.TrxCode  
   
 INSERT INTO @TempLines(TrxCode,LineCount)  
 SElECT TH.TrxCode,COUNT(TD.[LineNo])  
 FROM tblCustomerVisit V           
 INNER JOIN tblCustomer C ON C.Code = V.ClientCode AND V.UserCode = @UserCode                
 INNER JOIN tblCustomerDetail CD On C.Code =CD.CustomerCode   
  AND (@ChannelCodes = '' OR CD.ChannelCode IN(SELECT Item FROM ufnSplit(@ChannelCodes, ',')))                 
 INNER JOIN tblTrxHeader TH ON TH.UserCode = V.UserCode AND TH.VisitCode = V.VisitCode                  
  AND V.ClientCode = TH.ClientCode  
  AND TH.OrgCode = CD.SalesOrgCode       
  AND TH.TrxType IN (1,3,5) AND TH.TrxStatus IN (100,200,3)            
  AND DATEDIFF(dd, TH.TrxDate, V.Date) = 0  
 INNER JOIN tblTrxDetail TD ON TH.TrxCode = TD.TrxCode  
 WHERE (DATEDIFF(dd,@StartDate,V.[Date]) >= 0 AND DATEDIFF(dd,@EndDate,V.[Date]) <= 0)                 
 GROUP BY TH.TrxCode  
   
 SELECT @LinesperPC = ISNULL(SUM(LineCount),0) FROM @TempLines  
   
 UPDATE T  
 SET T.ProductiveCalls = @ProductiveCalls,  
  T.AvgValuePerPC = (CASE WHEN @ProductiveCalls = 0 THEN 0 ELSE @AvgValuePerPC / @ProductiveCalls END),  
  T.LinesperPC = (CASE WHEN @ProductiveCalls = 0 THEN 0 ELSE @LinesperPC / @ProductiveCalls END),  
  T.MaxLineinPC = (SELECT ISNULL(MAX(LineCount),0) FROM @TempLines),  
  T.MinLineinPC = (SELECT ISNULL(MIN(LineCount),0) FROM @TempLines)  
 FROM @TemTable T  
 WHERE T.UserCode = @UserCode   
   
 SET @CurrId = @CurrId + 1  
 SET @Count = @Count + 1  
   
END  
  
SELECT UserCode,UserName,ProductiveCalls,AvgValuePerPC,LinesperPC,MaxLineinPC,MinLineinPC FROM @TemTable  
  
SELECT COUNT(1) AS BillingEfficiencyCount FROM @TemTable
