BEGIN              
      
           
    
select S.Code UserCode INTO #UserHierarchy     
from Vw_SalesUserHierarchy    S    
where (ISNULL(@Salesman,'')='' OR S.Code in (select Items from dbo.Split(@Salesman,',')))    
and  (ISNULL(@Supervisor,'')='' OR S.Supervisor in (select Items from dbo.Split(@Supervisor,',')))    
and  (ISNULL(@RSMs,'')='' OR S.ASMCode in (select Items from dbo.Split(@RSMs,',')))    
and  (ISNULL(@SalesExecutive,'')='' OR S.SalesExecutive in (select Items from dbo.Split(@SalesExecutive,',')))    

select Code,ParentSalesOrg INTO #SalesOrgData from tblSalesOrganization      




Select * into #RouteSalesSummaryByItemCustomer From tblRouteSalesSummaryByItemCustomer  ER Where MONTH(ER.[Date])=MONTH(@StartDate) and YEAR(ER.[Date]) =YEAR(@StartDate)      



select T.ItemCode,I.Description as ItemName,Round(GrossSales,2) GrossSales       
,0 Target,0 Variance,Round(MTD_GrossSales,2) MTD_GrossSales,0 MTD_Target,ROUND(MTD_Wastage,2) MTD_Wastage      
from (      
select ItemCode      
,SUM(case WHEN Datediff(dd,@StartDate,[Date])=0 THEN TotalSales ELSE 0 END) GrossSales       
,SUM(TotalSales) MTD_GrossSales      
,SUM(TotalGRSales+TotalDamageSales+TotalExpirySales) MTD_Wastage      
from #RouteSalesSummaryByItemCustomer ER  
  
inner join #UserHierarchy  U on U.UserCode=ER.UserCode      
Inner join tblRoute R on R.Code=ER.RouteCode      
Inner join #SalesOrgData S on S.Code=R.SalesOrgCode      
Inner join tblItem I on I.Code=ER.ItemCode and I.SalesOrgCode=S.ParentSalesOrg       
Inner join tblCustomerDetail CD on CD.CustomerCode=ER.CustomerCode and CD.SalesOrgCode=R.SalesOrgCode      
where    
 (ISNULL(@SalesOrgCodes,'')='' OR S.Code in (select Items from dbo.Split(@SalesOrgCodes,',') ))      
and (ISNULL(@Route,'')='' OR ER.RouteCode in (select Items from dbo.Split(@Route,',') ))      
and (ISNULL(@Category,'')='' OR I.GroupLevel4 in (select Items from dbo.Split(@Category,',') ))      
and (ISNULL(@Brand,'')='' OR I.GroupLevel2 in (select Items from dbo.Split(@Brand,',') ))      
and (ISNULL(@SubBrand,'')='' OR I.GroupLevel3 in (select Items from dbo.Split(@SubBrand,',') ))      
and (ISNULL(@Channel,'')='' OR CD.ChannelCode in (select Items from dbo.Split(@Channel,',') ))      
and (ISNULL(@SubChannel,'')='' OR CD.SubChannelCode in (select Items from dbo.Split(@SubChannel,',') ))
and MONTH(ER.[Date])=MONTH(@StartDate) and YEAR(ER.[Date]) =YEAR(@StartDate)      
group by ItemCode      
) as T      
inner join (select distinct Code,Description from tblItem) I on I.Code = T.ITemCode      
Order by 1      

Drop table #SalesOrgData
Drop table #UserHierarchy
Drop table #RouteSalesSummaryByItemCustomer
